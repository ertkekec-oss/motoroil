model NetworkLiquidityOpportunity {
  id             String  @id @default(cuid())
  categoryId     String?
  productRef     String?
  regionCode     String?
  supplyTenantId String?
  demandTenantId String?
  clusterId      String?

  opportunityType NetworkLiquidityOpportunityType

  confidenceScore      Float @default(0)
  matchScore           Float @default(0)
  liquidityVolumeScore Float @default(0)

  routingEligibilityScore  Float @default(0)
  shippingFeasibilityScore Float @default(0)
  financialSafetyScore     Float @default(0)

  status NetworkLiquidityOpportunityStatus @default(DISCOVERED)

  expiresAt          DateTime?
  dedupeKey          String    @unique
  calculationVersion String

  explainJson Json?

  createdAt DateTime @default(now())
  updatedAt DateTime @updatedAt

  matches NetworkLiquidityMatch[]

  @@index([categoryId, regionCode])
  @@index([status, expiresAt])
  @@index([supplyTenantId])
  @@index([demandTenantId])
  @@index([matchScore])
}

model NetworkLiquidityMatch {
  id            String                      @id @default(cuid())
  opportunityId String
  opportunity   NetworkLiquidityOpportunity @relation(fields: [opportunityId], references: [id], onDelete: Cascade)

  buyerTenantId  String
  sellerTenantId String

  productRef String?
  categoryId String?

  graphDistance   Int   @default(0)
  trustScore      Float @default(0)
  reputationScore Float @default(0)

  shippingReliabilityScore Float @default(0)
  financialRiskScore       Float @default(0)

  estimatedPriceRangeLow  Float?
  estimatedPriceRangeHigh Float?

  finalMatchScore Float @default(0)

  status NetworkLiquidityMatchStatus @default(CANDIDATE)

  explainJson Json?

  createdAt DateTime @default(now())
  updatedAt DateTime @updatedAt

  @@index([buyerTenantId])
  @@index([sellerTenantId])
  @@index([opportunityId])
  @@index([status])
}

model NetworkTradeProposal {
  id               String @id @default(cuid())
  opportunityId    String
  liquidityMatchId String

  buyerTenantId  String
  sellerTenantId String

  categoryId String?
  productRef String?

  proposedQuantityLow  Int @default(0)
  proposedQuantityHigh Int @default(0)

  proposedPriceLow  Float @default(0)
  proposedPriceHigh Float @default(0)

  currency String @default("TRY")

  shippingMode   String  @default("STANDARD")
  paymentMode    String  @default("ESCROW")
  escrowRequired Boolean @default(true)

  status     NetworkTradeProposalStatus @default(DRAFT)
  policyMode NetworkTradePolicyMode     @default(SUGGEST_ONLY)

  confidenceScore Float @default(0)
  riskScore       Float @default(0)

  expiresAt          DateTime?
  dedupeKey          String    @unique
  calculationVersion String

  createdAt DateTime @default(now())
  updatedAt DateTime @updatedAt

  messages  NetworkTradeProposalMessage[]
  decisions NetworkTradeProposalDecision[]

  @@index([buyerTenantId, status])
  @@index([sellerTenantId, status])
  @@index([opportunityId])
  @@index([liquidityMatchId])
  @@index([status, expiresAt])
}

model NetworkTradeProposalMessage {
  id         String               @id @default(cuid())
  proposalId String
  proposal   NetworkTradeProposal @relation(fields: [proposalId], references: [id], onDelete: Cascade)

  senderTenantId String

  messageType NetworkTradeProposalMessageType @default(COMMENT)

  payloadJson Json?

  createdAt DateTime @default(now())

  @@index([proposalId])
  @@index([senderTenantId])
}

model NetworkTradeProposalDecision {
  id         String               @id @default(cuid())
  proposalId String
  proposal   NetworkTradeProposal @relation(fields: [proposalId], references: [id], onDelete: Cascade)

  decisionType NetworkTradeProposalDecisionType

  decidedByTenantId String?
  autoPolicy        Boolean @default(false)

  decisionReason String?

  createdAt DateTime @default(now())

  @@index([proposalId])
}
